document.body.addEventListener('click', handleBodyClick); function handleBodyClick(event) { const target = event.target; // 隐藏 contextmenu const contextMenu = document.getElementById('custom-context-menu'); if (contextMenu && !contextMenu.contains(target)) { hideContextMenu(); } // 隐藏 theme-switcher__options const themeSwitcherOptions = document.querySelector('.theme-switcher__options.show'); const themeSwitcherButton = document.querySelector('.theme-switcher-option'); if ( themeSwitcherOptions && !themeSwitcherOptions.contains(target) && (!themeSwitcherButton || !themeSwitcherButton.contains(target)) ) { hideThemeSwitcherOptions(); } } function hideContextMenu() { const contextMenu = document.getElementById('custom-context-menu'); if (contextMenu) { contextMenu.remove(); } } function hideThemeSwitcherOptions() { const themeSwitcherOptions = document.querySelector('.theme-switcher__options.show'); if (themeSwitcherOptions) { themeSwitcherOptions.classList.remove('show'); } }
